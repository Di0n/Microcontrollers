#define PULSE_OUT 0
#define PULSE_IN 1
// Pulse out = 0
// Pulse in = 1
void write(int val);

volatile int pulse = 0;

ISR(INT0_vect)
{
	static int i = 0;
	if (i == 1)
	{
		TCCR1B = 0;
		pulse = TCNT1;
		TCNT1 = 0;
		i = 0;
	}
	if (i == 0)
	{
		TCCR1B |= 1 << CS10;
		i = 1;
	}
}

{
	// 0 = input 1 = output

	LCD_Init();
    while (1) 
    {
		write(1);
		Utils_Wait(15);
		write(0);

		pulse = PORTA1;
		int count = pulse / 58;
		LCD_ClearDisplay();
		LCD_SetCursorPos(0x00);
		
		char str[8];
		itoa(count, str, 8);
		LCD_WriteString(str);
		
		//Utils_Wait(10);
    }

	return 0;
}

void write(int toggle)
{
	if (toggle == 1)
		PORTA |= (1<<PULSE_OUT);
	else if (toggle == 0)
		PORTA &= ~(1<<PULSE_OUT);
}
